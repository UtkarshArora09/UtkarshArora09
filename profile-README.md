<script>
const canvas = document.getElementById('board');
const ctx = canvas.getContext('2d');
const SIZE = 500, CELLS = 10, CELL = SIZE / CELLS;
const SNAKES = { 97:78, 95:56, 88:24, 62:19, 36:6, 32:10, 49:11, 17:7 };
const LADDERS = { 4:14, 9:31, 20:38, 28:84, 40:59, 51:67, 63:81, 71:91 };
const DICE_FACES = ['', '⚀','⚁','⚂','⚃','⚄','⚅'];
let players = [{ pos: 0, color: '#FF6B6B', name: 'Player 1' }, { pos: 0, color: '#4ECDC4', name: 'Player 2' }];
let turn = 0, rolling = false;

function cellToXY(cell) {
  if (cell === 0) return { x: -999, y: -999 };
  cell--;
  const row = Math.floor(cell / CELLS);
  const col = cell % CELLS;
  const x = (row % 2 === 0) ? col : (CELLS - 1 - col);
  const y = CELLS - 1 - row;
  return { x: x * CELL + CELL / 2, y: y * CELL + CELL / 2 };
}

function drawBoard() {
  for (let r = 0; r < CELLS; r++) {
    for (let c = 0; c < CELLS; c++) {
      const num = (CELLS - 1 - r) * CELLS + ((r % 2 === 0) ? c + 1 : CELLS - c);
      const px = c * CELL, py = r * CELL;
      ctx.fillStyle = (r + c) % 2 === 0 ? '#1e1e3f' : '#16213e';
      ctx.fillRect(px, py, CELL, CELL);
      ctx.strokeStyle = '#ffffff11'; ctx.strokeRect(px, py, CELL, CELL);
      ctx.fillStyle = '#ffffff33'; ctx.font = `bold ${CELL * 0.22}px Segoe UI`;
      ctx.textAlign = 'center'; ctx.textBaseline = 'top';
      ctx.fillText(num, px + CELL / 2, py + 4);
    }
  }
}

function drawSnakesAndLadders() {
  // Draw Snakes
  Object.entries(SNAKES).forEach(([from, to]) => {
    const a = cellToXY(+from), b = cellToXY(+to);
    ctx.beginPath();
    ctx.moveTo(a.x, a.y);
    const mx = (a.x + b.x) / 2 + 20, my = (a.y + b.y) / 2;
    ctx.quadraticCurveTo(mx, my, b.x, b.y);
    ctx.strokeStyle = '#FF6B6B'; ctx.lineWidth = 5; ctx.lineCap = 'round'; ctx.stroke();
    ctx.fillStyle = '#FF6B6B'; ctx.font = '16px serif';
    ctx.fillText('🐍', a.x - 10, a.y - 10);
  });
  // Draw Ladders
  Object.entries(LADDERS).forEach(([from, to]) => {
    const a = cellToXY(+from), b = cellToXY(+to);
    ctx.beginPath();
    ctx.moveTo(a.x, a.y); ctx.lineTo(b.x, b.y);
    ctx.strokeStyle = '#4ECDC4'; ctx.lineWidth = 5; ctx.lineCap = 'round'; ctx.stroke();
    ctx.fillStyle = '#4ECDC4'; ctx.font = '16px serif';
    ctx.fillText('🪜', b.x - 10, b.y - 10);
  });
}

function drawPlayers() {
  players.forEach((p, i) => {
    if (p.pos === 0) return;
    const { x, y } = cellToXY(p.pos);
    const offset = i === 0 ? -10 : 10;
    ctx.beginPath();
    ctx.arc(x + offset, y, 12, 0, Math.PI * 2);
    ctx.fillStyle = p.color; ctx.fill();
    ctx.strokeStyle = '#fff'; ctx.lineWidth = 2; ctx.stroke();
    ctx.fillStyle = '#fff'; ctx.font = 'bold 10px Segoe UI';
    ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
    ctx.fillText(i + 1, x + offset, y);
  });
}

function render() {
  ctx.clearRect(0, 0, SIZE, SIZE);
  drawBoard(); drawSnakesAndLadders(); drawPlayers();
}

function log(msg) {
  const el = document.getElementById('log');
  const d = document.createElement('div');
  d.textContent = msg; el.prepend(d);
}

function updateUI() {
  document.getElementById('p1pos').textContent = `Position: ${players[0].pos}`;
  document.getElementById('p2pos').textContent = `Position: ${players[1].pos}`;
  document.getElementById('turnBadge').textContent = `${players[turn].name}'s Turn`;
}

async function rollDice() {
  if (rolling) return;
  rolling = true;
  document.getElementById('rollBtn').disabled = true;
  // Animate dice roll
  for (let i = 0; i < 8; i++) {
    document.getElementById('diceDisplay').textContent = DICE_FACES[Math.ceil(Math.random() * 6)];
    await new Promise(r => setTimeout(r, 80));
  }
  const dice = Math.ceil(Math.random() * 6);
  document.getElementById('diceDisplay').textContent = DICE_FACES[dice];
  const p = players[turn];
  let newPos = p.pos + dice;
  log(`${p.name} rolled ${dice} → moved to ${Math.min(newPos, 100)}`);
  if (newPos > 100) { log(`${p.name} needs exact! Stays at ${p.pos}`); rolling = false; document.getElementById('rollBtn').disabled = false; turn = 1 - turn; updateUI(); return; }
  p.pos = newPos;
  if (SNAKES[p.pos]) { log(`😬 ${p.name} hit a snake! ${p.pos} → ${SNAKES[p.pos]}`); p.pos = SNAKES[p.pos]; }
  else if (LADDERS[p.pos]) { log(`🎉 ${p.name} climbed a ladder! ${p.pos} → ${LADDERS[p.pos]}`); p.pos = LADDERS[p.pos]; }
  render(); updateUI();
  if (p.pos === 100) {
    document.getElementById('winnerText').textContent = `🎉 ${p.name} Wins!`;
    document.getElementById('winnerModal').classList.add('show');
    rolling = false; return;
  }
  turn = 1 - turn; updateUI();
  rolling = false; document.getElementById('rollBtn').disabled = false;
}

function resetGame() {
  players[0].pos = 0; players[1].pos = 0; turn = 0;
  document.getElementById('log').innerHTML = '';
  document.getElementById('diceDisplay').textContent = '🎲';
  document.getElementById('rollBtn').disabled = false;
  rolling = false; render(); updateUI();
  log('🎮 New game started!');
}

render(); updateUI(); log('🎮 Game started! Player 1 goes first.');
